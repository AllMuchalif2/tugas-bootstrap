Database

  • Membuat Database

    create database nama_database;

  • Melihat Database

    show databases;

  • Mengaktifkan Database

    use nama_database;

  • Menghapus Database

    drop database nama_database;

Tabel

  • Membuat Tabel 
    create table nama_tabel(
Nama_field1 Tipe_data (ukuran) primary key not null,
 Nama_field2 Tipe_data (ukuran)not null,...) ;
  • Melihat Daftar Tabel

    show tables;

  • Melihat Struktur Tabel

    desc nama_table;

    describe nama_table;

    explain nama_table;

    show columns from nama_table;

  • Mengganti Nama Tabel

    rename table nama_table_lama to nama_tabel_baru;

  • Menghapus Tabel

    drop table nama_tabel;

Manipulasi Tabel

  • Menambah Field pada Tabel

    alter table nama_tabel add nama_field type_data;

  • Mengganti Nama Field pada Tabel

    alter table nama_tabel change field_lama field_baru type_data;

  • Mengubah Type Data Field pada Tabel

    alter table nama_tabel modify nama_field type_data_baru(panjang baru);

  • Menghapus Field pada Tabel

    alter table nama_tabel drop nama_field;

Manipulasi Data

  • Memasukkan Data pada Tabel 
    insert into nama_tabel values 
('isi_field_1','isi_field_2','...','isi_field_n');
    insert into nama_tabel(field_1,field_2,...,field_n) values 
('isi_field_1','isi_field_2','...','isi_field_n');
  • Melihat Isi Record

    select * from nama_tabel;

  • Memperbarui Isi Data

    update nama_tabel set field_1='data_baru';

    update nama_tabel set field_1='data_baru' where [kondisi];

  • Menghapus Isi Data

    delete from nama_tabel;

    delete from nama_tabel where [kondisi];

Seleksi Data

  • Menampilkan Data dengan where

    select field_1,field_2,...,field_n from nama_table;

  • Menampilkan Data dengan Between 
    select field_1,field_2,...,field_n  
from nama_table
where [kondisi]
between 'nilai_1' and 'nilai_2';
  • Menggunakan Operator Like 
    select nama_field  
from nama_table
where nama_field 
like 'karakter%';
  • Membatasi Data dengan Limit 
    select nama_field  
from nama_table
limit batasan ;
  • Seleksi Data dengan Klausa Where 
    select nama_field  
from nama_table
where kondisi ;
    • Operator Keterangan
    = Sama dengan
    !=, Tidak sama dengan
    > Lebih besar
    < Kurang dari
    >= Lebih besar sama dengan
    <= Kurang dari sama dengan
  • Menggunakan Operator Logika 
    select nama_field  
from nama_table
where kondisi ;
    • Operator Keterangan
    and , && data benar bila kedua pembanding bernilai benar
    or , || data bernilai benar apabila salah satu data atau keduanya bernilai benar
    not , ! data dianggap benar apabila bernilai salah/ tidak sesuai dengan nilai pembanding
    xor Data dianggap benar jika hanya data pertama yang bernialai pertama yang bernailai benar
  • Menggunakan Operator Logika 
    select distinct nama_field  
from nama_table;
